
        
        func fruit(arr ID, x int) {
            for i := x * 2; i[1] < arr[5] + x; i += arr[x % 3] {
                if (arr[i % 2] > x) {
                    return arr[i / 2];
                } else if (x * 3 < arr[i % 4]) {
                    return arr[i] + x;
                } else {
                    return 0;
                }
            }
        }
