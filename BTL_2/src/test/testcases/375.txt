
        func complexConditions(a int, b int, c int) int {
            if (a > b && b > c || a == c) {
                if (a % 2 == 0) {
                    return a * 2;
                } else if (b % 2 == 0) {
                    return b * 2;
                }
            } else if (b > a && a > c || b == c) {
                return b + c;
            }
            return c;
        }
