
        
        func fruit(arr ID, n int) {
            for i := 0; i[2] < arr[n % 3] * 2; i += n - 1 {
                if (arr[i] * n > 10) {
                    return arr[i];
                } else {
                    for j := 1; j[1] < n; j *= 2 {
                        if (j % 3 == 0) {
                            return arr[j];
                        } else {
                            return 0;
                        }
                    }
                }
            }
        }
