
        func hi(a int, b int, c int, d int, e int) int {
            if (a > b && (c > d || e > a) && !(a == c)) {
                if (a % 2 == 0 && b % 2 == 0) {
                    if (c % 3 == 0 || d % 3 == 0) {
                        if (e % 5 == 0) {
                            return a * b + c * d - e;
                        } else if (e % 4 == 0) {
                            if (a > c + d) {
                                return a - b + c;
                            } else {
                                return b - a + d;
                            }
                        }
                    } else if (c % 5 == 0 && d % 5 != 0) {
                        return c * d / (a + 1);
                    }
                } else if (a % 3 == 0 || b % 3 == 0) {
                    if (c * d > a * b && e < a + b + c) {
                        return (a + b) * (c - d) / (e + 1);
                    } else {
                        return (a - b) * (c + d) / (e - 1);
                    }
                }
            } else if (b > a && (d > c || e > b) && !(b == d)) {
                return b * c + a * d - e * e;
            } else if (c > a && c > b && (d > a || e > b)) {
                return c * c - a * b + d * e;
            } else {
                return (a + b + c + d + e) / 5;
            }
            return 0;
        }
